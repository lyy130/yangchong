1.1.1. 活动弹窗

  • 请求地址: /activity/pop
  • 请求参数:
    'userId' int //用户id 必填
'userKey' string //用户key 必填

  • 返回结果:

    
 "result" //bool 返回状态
 "data"  // object
     "id" // int 编号
     "image" // string 图片地址,
     "button" // string 按钮地址,
     "url": // string url地址,
     "type" // string 类型 "url" 时 取 url数据，'bigvdetail' 时 取 para数据,
     "para"// object 
      // 'bigvdetail'的数据 ['userId':1]

t_backend_activity_pop 弹窗类型 0活动 1主播详情 2跳转房间 3心动匹配 4全屏h5

1.1.3. 集物榜 -- 二级标签

  • 请求地址: /activity/gather-gift-tab
  • 请求参数:

'userKey' string //用户key 必填

'data' object

1.1.7. 集物榜表结构

ALTER TABLE `t_backend_v42_activity_list` 
    MODIFY COLUMN `type` tinyint(4) NOT NULL DEFAULT 1 COMMENT '类型  1专题活动  2周星榜 3 不知道 4是游戏中心 5是半屏游戏 6是全屏游戏 7是集物榜 8是全屏web' AFTER `end_date`;

CREATE TABLE `t_gather_gift_config` (
    `id` int(11) unsigned NOT NULL AUTO_INCREMENT,
    `act_id` int(4) unsigned NOT NULL DEFAULT '1' COMMENT '关联活动主表编号',
    `chat_act_id` int(4) unsigned NOT NULL DEFAULT '0' COMMENT '关联房间活动主表编号,需要开房间活动时可使用',
    `get_gift_id` varchar(128) CHARACTER SET utf8mb4 NOT NULL DEFAULT '' COMMENT '获得的礼物编号 例：1,2',
    `giving_gift_id` int(11) unsigned NOT NULL DEFAULT '0' COMMENT '赠送的礼物编号',
    `giving_gift_series` varchar(32) CHARACTER SET utf8mb4 NOT NULL DEFAULT '' COMMENT '赠送的礼物系列名称',
    `exclude_id` varchar(128) CHARACTER SET utf8mb4 NOT NULL DEFAULT '' COMMENT '需要排除榜单的用户编号',
    `limit_count` tinyint(4) unsigned NOT NULL DEFAULT '0' COMMENT '进行中榜单记录条数',
    `end_limit_count` tinyint(4) unsigned NOT NULL DEFAULT '0' COMMENT '结束后榜单记录条数',
    `is_default` tinyint(4) unsigned NOT NULL DEFAULT '0' COMMENT '是否是默认页',
    `type` tinyint(4) unsigned NOT NULL DEFAULT '0' COMMENT '配置类型 0集物榜配置， 1专题',
    `effect_notice_id` tinyint(4) unsigned NOT NULL DEFAULT '0' COMMENT '特效编号，t_backend_global_effect_notice表主键',
    PRIMARY KEY (`id`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8 COMMENT='集物榜配置';

CREATE TABLE `t_gather_gift_log` (
    `id` int(11) unsigned NOT NULL AUTO_INCREMENT,
    `user_id` int(11) unsigned NOT NULL DEFAULT '0' COMMENT '用户编号',
    `gift_id` int(11) NOT NULL DEFAULT '0' COMMENT '获得的礼物编号',
    `vcoin` int(11) NOT NULL DEFAULT '0' COMMENT '礼物价格',
    `create_time` datetime NOT NULL DEFAULT '0000-00-00 00:00:00' COMMENT '创建时间',
    `use_time` datetime NOT NULL DEFAULT '0000-00-00 00:00:00' COMMENT '使用时间',
    `is_use` tinyint(4) unsigned NOT NULL DEFAULT '0' COMMENT '是否已使用 0未使用 1已使用',
    PRIMARY KEY (`id`),
    KEY `idx_user_id` (`user_id`) USING BTREE,
    KEY `idx_gift_id` (`gift_id`) USING BTREE
) ENGINE=InnoDB DEFAULT CHARSET=utf8 COMMENT='集物榜礼物收集日志';

CREATE TABLE `t_gather_gift_get_reward_log` (
    `id` int(11) unsigned NOT NULL AUTO_INCREMENT,
    `act_id` int(4) unsigned NOT NULL DEFAULT '1' COMMENT '关联活动主表编号',
    `user_id` int(11) unsigned NOT NULL DEFAULT '0' COMMENT '用户编号',
    `log_ids` varchar(128) CHARACTER SET utf8mb4 NOT NULL DEFAULT '' COMMENT '关联t_gather_gift_log表主机 例：1,2',
    `create_time` datetime NOT NULL DEFAULT '0000-00-00 00:00:00' COMMENT '创建时间',
    `accept_time` datetime NOT NULL DEFAULT '0000-00-00 00:00:00' COMMENT '派奖完成时间',
    PRIMARY KEY (`id`),
    KEY `idx_user_id` (`user_id`) USING BTREE,
    KEY `idx_act_id` (`act_id`) USING BTREE
) ENGINE=InnoDB DEFAULT CHARSET=utf8 COMMENT='集物榜礼物收集日志';
